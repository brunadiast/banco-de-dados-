create table funcionario(pk_funcionario serial primary key,
			fk_cargo int,
			nome char(55),
			cpf varchar(11),
			nascimento date,
			salario decimal (10,2),
			constraint fk_cargo foreign key (fk_cargo) references 
				cargo(pk_cargo))
---------------------------------------------------------------------------------------------------------------------------------------------------
select * from funcionario

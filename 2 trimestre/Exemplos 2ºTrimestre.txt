Mostrando duas tabelas de modo perfeito ao mesmo tempo:

Modo interno:

SELECT * FROM aluno JOIN aluno_endereco ON pk_aluno = fk_aluno

OU

SELECT * FROM aluno INNER JOIN aluno_endereco ON pk_aluno = fk_aluno

Modo externo:

SELECT * FROM aluno LEFT OUTER JOIN aluno_endereco ON pk_aluno = fk_aluno

OU

SELECT * FROM aluno RIGHT OUTER JOIN aluno_endereco ON pk_aluno = fk_aluno

OU

SELECT * FROM aluno FULL OUTER JOIN aluno_telefone ON pk_aluno = fk_aluno
